21const W: usize = 240; // Width of the display
22const H: usize = 280; // Height of the display
23
24static BUFFER: &[u8] = include_bytes!("../output.rgb"); // RGB565 data for the display
25// static BUFFER: [u8; W * H * 2] = [0u8; W * H * 2]; // test black image
26
27#[entry]
28fn main() -> ! {
29    let dp = stm32f4xx_hal::pac::Peripherals::take().unwrap();
30    let cp = cortex_m::Peripherals::take().unwrap();
31
32    let mut syst = cp.SYST;
33
34    let rcc = dp.RCC.constrain();
35
36    let clocks = rcc
37        .cfgr
38        .use_hse(25.MHz())
39        .sysclk(48.MHz())
40        .hclk(48.MHz())
41        .freeze();
42
43    let sfreq = clocks.sysclk().raw();
44    let hfreq = clocks.hclk().raw();
45
46    let src = syst.get_clock_source();
47
48    if let SystClkSource::Core = src {
49        info!("using Core clock source"); // rtt is epic
50    } else {
51        info!("using external clock source");
52    }
53
54    info!("sysclk:{}\thclk:{}", sfreq, hfreq);
55
56    let pa = dp.GPIOA.split();
57
58    let pa7_mosi = pa
59        .pa7
60        .into_push_pull_output()
61        .speed(Speed::VeryHigh)
62        .into_alternate();
63    let false_pin = gpio::NoPin::new();
64    let pa5_sck = pa
65        .pa5
66        .into_push_pull_output()
67        .speed(Speed::VeryHigh)
68        .into_alternate();
69
70    let mode = spi::Mode {
71        polarity: Polarity::IdleHigh,
72        phase: Phase::CaptureOnSecondTransition,
73    };
74    let spi = Spi::new(
75        dp.SPI1,
76        (pa5_sck, false_pin, pa7_mosi),
77        mode,
78        16.MHz(),
79        &clocks,
80    );
81    let dc = pa.pa4.into_push_pull_output().speed(Speed::High); // high for data and low for command
82    let cs = pa.pa3.into_push_pull_output().speed(Speed::High);
83    let rst = pa.pa2.into_push_pull_output().speed(Speed::High);
84
85    let mut d = Delay::new(syst, clocks.hclk().raw());
86    let mut st7789v2: ST7789V2<
87        _, // SPI type
88        _, // DC pin type
89        _, // RST pin type
90        _, // CS pin type
91        W, // width
92        H, // height
93    > = ST7789V2::new(spi, dc, rst, cs, &mut d);
94
95    st7789v2
96        .init()
97        .expect("Failed to initialize ST7789V2 display");
98
99    // let bmp_data = include_bytes!("../testimage.bmp");
100    // let _bmp: Bmp<Rgb565> = Bmp::from_slice(bmp_data).expect("Failed to load BMP image");
101
102    st7789v2
103        .send_command(Commands::InversionOn)
104        .expect("Failed to enable inversion"); // Enable inversion since the display is inverted by default
105    st7789v2.draw_screen(BUFFER).expect("Failed to draw screen");
106    loop {}
107}
